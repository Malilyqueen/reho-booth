  <script>
    function format(amount) {
      return `AED ${amount.toFixed(2).replace('.', ',')}`;
    }

    function recalculateTotals() {
      let total = 0;

      document.querySelectorAll('.category').forEach(cat => {
        let catTotal = 0;
        cat.querySelectorAll('.subcategory').forEach(sub => {
          let subTotal = 0;
          sub.querySelectorAll('.line-amount').forEach(input => {
            const val = parseFloat(input.value) || 0;
            subTotal += val;
          });
          sub.querySelector('.sub-total').textContent = format(subTotal);
          catTotal += subTotal;
        });
        cat.querySelector('.cat-total').textContent = format(catTotal);
        total += catTotal;
      });

      document.getElementById('totalBudget').textContent = format(total);
    }

    function setupEvents() {
      document.querySelectorAll('.line-amount').forEach(input => {
        input.addEventListener('input', recalculateTotals);
      });

      document.querySelectorAll('.add-line').forEach(btn => {
        btn.addEventListener('click', () => {
          const container = btn.parentElement;
          const line = document.createElement('div');
          line.className = 'line';
          line.innerHTML = `
            <input type="text" placeholder="Dépense" class="line-label" />
            <input type="number" value="0" class="line-amount" /> AED
          `;
          container.insertBefore(line, btn);
          line.querySelector('.line-amount').addEventListener('input', recalculateTotals);
        });
      });

      document.querySelectorAll('.add-subcategory').forEach(btn => {
        btn.addEventListener('click', () => {
          const sub = document.createElement('div');
          sub.className = 'subcategory';
          sub.innerHTML = `
            <h4>Nouvelle sous-cat - <span class="sub-total">AED 0,00</span></h4>
            <div class="line">
              <input type="text" placeholder="Dépense" class="line-label" />
              <input type="number" value="0" class="line-amount" /> AED
            </div>
            <button class="add-line">+ Ajouter une ligne</button>
          `;
          btn.before(sub);
          setupEvents();
          recalculateTotals();
        });
      });
    }

    document.addEventListener('DOMContentLoaded', () => {
      setupEvents();
      recalculateTotals();
    });
  </script>
